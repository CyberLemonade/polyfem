281  {
282  POLYFEM_SCOPED_TIMER("line search", line_search_time);
283 
284  if (!m_line_search)

287  double rate = m_line_search->line_search(x, delta_x, objFunc);
288

292  polyfem::utils::logger().warn(

294  this->m_status = Status::Continue; // Try the step again with gradient descent
295  }
296  else if (std::isnan(rate))
297  {
298  assert(descent_strategy == 2); // failed on gradient descent
299  polyfem::utils::logger().error("[{}] Line search failed on gradient descent; stopping", name());

301  throw std::runtime_error("Line search failed on gradient descent");
302  }
