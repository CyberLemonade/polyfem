16  // For the purpose of the tagging, elements (facets in 2D, cells in 3D) adjacent to a polytope
17  // are tagged as boundary, and vertices incident to a polytope are also considered as boundary.
18  enum class ElementType
19  {
20  Simplex, // Triangle/tet element
21  RegularInteriorCube, // Regular quad/hex inside a 3^n patch
22  SimpleSingularInteriorCube, // Quad/hex incident to exactly 1 singular vertex (in 2D) or edge (in 3D)
23  MultiSingularInteriorCube, // Quad/Hex incident to more than 1 singular vertices (should not happen in 2D)
24  RegularBoundaryCube, // Boundary quad/hex, where all boundary vertices/edges are incident to at most 2 quads/hexes
25  SimpleSingularBoundaryCube, // Quad incident to exactly 1 singular vertex (in 2D); hex incident to exactly 1 singular interior edge, 0 singular boundary edge, 1 boundary face (in 3D)
26  MultiSingularBoundaryCube, // Boundary hex that is not regular nor SimpleSingularBoundaryCube
27  InterfaceCube, // Quad/hex that is at the interface with a polytope (if a cube has both external boundary and and interface with a polytope, it is marked as interface)
28  InteriorPolytope, // Interior polytope
29  BoundaryPolytope, // Boundary polytope
30  Undefined // For invalid configurations
31  };
