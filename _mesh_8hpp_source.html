16 // NOTE:
17 // For the purpose of the tagging, elements (facets in 2D, cells in 3D) adjacent to a polytope
18 // are tagged as boundary, and vertices incident to a polytope are also considered as boundary.
19 enum class ElementType
20 {
21 Simplex, // Triangle/tet element
22 RegularInteriorCube, // Regular quad/hex inside a 3^n patch
23 SimpleSingularInteriorCube, // Quad/hex incident to exactly 1 singular vertex (in 2D) or edge (in 3D)
24 MultiSingularInteriorCube, // Quad/Hex incident to more than 1 singular vertices (should not happen in 2D)
25 RegularBoundaryCube, // Boundary quad/hex, where all boundary vertices/edges are incident to at most 2 quads/hexes
26 SimpleSingularBoundaryCube, // Quad incident to exactly 1 singular vertex (in 2D); hex incident to exactly 1 singular interior edge, 0 singular boundary edge, 1 boundary face (in 3D)
27 MultiSingularBoundaryCube, // Boundary hex that is not regular nor SimpleSingularBoundaryCube
28 InterfaceCube, // Quad/hex that is at the interface with a polytope (if a cube has both external boundary and and interface with a polytope, it is marked as interface)
29 InteriorPolytope, // Interior polytope
30 BoundaryPolytope, // Boundary polytope
31 Undefined // For invalid configurations
32 };
