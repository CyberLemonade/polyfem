126  // https://github.com/ethz-asl/map_api/blob/master/map-api-common/include/map-api-common/eigen-hash.h
127  template <typename Scalar, int Rows, int Cols>
128  struct hash<Eigen::Matrix<Scalar, Rows, Cols>>
129  {
130  // https://wjngkoh.wordpress.com/2015/03/04/c-hash-function-for-eigen-matrix-and-vector/
131  size_t operator()(const Eigen::Matrix<Scalar, Rows, Cols> &matrix) const
132  {
133  size_t seed = 0;
134  for (size_t i = 0; i < matrix.size(); ++i)
135  {
136  Scalar elem = *(matrix.data() + i);
137  seed ^=
138  std::hash<Scalar>()(elem) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
139  }
140  return seed;
141  }
142  };
